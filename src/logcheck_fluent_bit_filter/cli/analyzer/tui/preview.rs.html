1//! Preview widget for analyzer TUI
2
3use ratatui::{
4    layout::Rect,
5    style::{Color, Modifier, Style},
6    text::{Line, Span},
7    widgets::{Block, Borders, Paragraph},
8    Frame,
9};
10use regex::Regex;
11
12pub struct Preview<'a> {
13    entries: &'a [String],
14    scroll: usize,
15    total_matches: usize,
16    pattern: Option<&'a Regex>,
17    focused: bool,
18    wrap: bool,
19    h_scroll: usize,
20}
21
22impl<'a> Preview<'a> {
23    pub fn new(
24        entries: &'a [String],
25        scroll: usize,
26        total_matches: usize,
27        pattern: Option<&'a Regex>,
28        focused: bool,
29        wrap: bool,
30        h_scroll: usize,
31    ) -> Self {
32        Self {
33            entries,
34            scroll,
35            total_matches,
36            pattern,
37            focused,
38            wrap,
39            h_scroll,
40        }
41    }
42
43    pub fn render(&self, f: &mut Frame, area: Rect) {
44        let visible_height = area.height.saturating_sub(2) as usize;
45        let lines = self
46            .entries
47            .iter()
48            .skip(self.scroll)
49            .take(visible_height)
50            .map(|entry| self.render_line(entry))
51            .collect::<Vec<_>>();
52
53        let title = format!(
54            "Preview ({} matches, PgUp/PgDn to scroll)",
55            self.total_matches
56        );
57        let block = Block::default()
58            .borders(Borders::ALL)
59            .title(title)
60            .border_style(self.border_style());
61        let mut widget = Paragraph::new(lines).block(block);
62        if self.wrap {
63            widget = widget.wrap(ratatui::widgets::Wrap { trim: false });
64        } else if self.h_scroll > 0 {
65            widget = widget.scroll((0, self.h_scroll as u16));
66        }
67        f.render_widget(widget, area);
68    }
69
70    fn render_line(&self, entry: &str) -> Line<'static> {
71        let Some(regex) = self.pattern else {
72            return Line::from(entry.to_string());
73        };
74
75        let highlight_style = Style::default()
76            .fg(Color::Black)
77            .bg(Color::Yellow)
78            .add_modifier(Modifier::BOLD);
79
80        let mut spans: Vec<Span> = Vec::new();
81        let mut last_end = 0;
82
83        for mat in regex.find_iter(entry) {
84            if mat.start() > last_end {
85                spans.push(Span::raw(entry[last_end..mat.start()].to_string()));
86            }
87            spans.push(Span::styled(mat.as_str().to_string(), highlight_style));
88            last_end = mat.end();
89        }
90
91        if spans.is_empty() {
92            return Line::from(entry.to_string());
93        }
94
95        if last_end < entry.len() {
96            spans.push(Span::raw(entry[last_end..].to_string()));
97        }
98
99        Line::from(spans)
100    }
101
102    fn border_style(&self) -> Style {
103        if self.focused {
104            Style::default()
105                .fg(Color::Cyan)
106                .add_modifier(Modifier::BOLD)
107        } else {
108            Style::default()
109        }
110    }
111}
