1use crate::cli::Result;
2use crate::cli::args::ShowMode;
3use crate::cli::output::{FilteredEntry, LogOutput};
4use serde_json;
5use std::io::{self, Write};
6
7/// JSON-based output formatter
8pub struct JsonOutput {
9    show_mode: ShowMode,
10    writer: Box<dyn Write>,
11}
12
13impl JsonOutput {
14    /// Create a new JSON output formatter that writes to stdout
15    pub fn new(show_mode: ShowMode) -> Self {
16        Self {
17            show_mode,
18            writer: Box::new(io::stdout()),
19        }
20    }
21
22    /// Create a new JSON output formatter with custom writer
23    pub fn with_writer(show_mode: ShowMode, writer: Box<dyn Write>) -> Self {
24        Self { show_mode, writer }
25    }
26}
27
28impl LogOutput for JsonOutput {
29    fn write_entry(&mut self, entry: &FilteredEntry) -> Result<()> {
30        if !entry.should_show(&self.show_mode) {
31            return Ok(());
32        }
33
34        // Create a JSON representation of the filtered entry
35        let json_entry = serde_json::json!({
36            "message": entry.message,
37            "matched": entry.matched,
38            "category": entry.category.as_ref().map(|c| format!("{:?}", c)),
39            "rule_type": match &entry.category {
40                Some(cat) => match cat {
41                    crate::rules::RuleCategory::Cracking => "cracking",
42                    crate::rules::RuleCategory::CrackingIgnore => "cracking_ignore",
43                    crate::rules::RuleCategory::Violations => "violations",
44                    crate::rules::RuleCategory::ViolationsIgnore => "violations_ignore",
45                    crate::rules::RuleCategory::SystemEvents => "ignore",
46                    crate::rules::RuleCategory::Workstation => "ignore",
47                    crate::rules::RuleCategory::Server => "ignore",
48                    crate::rules::RuleCategory::Local => "ignore",
49                },
50                None => "unmatched",
51            }
52        });
53
54        writeln!(self.writer, "{}", serde_json::to_string(&json_entry)?)?;
55        Ok(())
56    }
57
58    fn finish(&mut self) -> Result<()> {
59        self.writer.flush()?;
60        Ok(())
61    }
62}
