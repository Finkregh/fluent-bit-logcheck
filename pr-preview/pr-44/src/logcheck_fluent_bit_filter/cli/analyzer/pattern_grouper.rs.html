1/// Pattern grouping and regex generation using grex
2///
3/// This module groups similar log messages and generates
4/// regular expressions that match each group.
5use crate::cli::Result;
6use crate::regex_conversion;
7use anyhow::Context;
8use grex::RegExpBuilder;
9use regex::Regex;
10use std::collections::HashMap;
11
12/// A group of similar log messages with a generated regex
13#[derive(Debug, Clone)]
14pub struct PatternGroup {
15    /// The generated regex pattern in POSIX format (for logcheck)
16    pub regex: String,
17    /// Indices of messages that match this pattern
18    pub matching_indices: Vec<usize>,
19    /// Number of matches
20    pub match_count: usize,
21}
22
23impl PatternGroup {
24    /// Create a compiled regex from this pattern
25    pub fn compile(&self) -> Result<Regex> {
26        // Convert POSIX to Rust format for compilation
27        let rust_pattern = regex_conversion::posix_to_rust(&self.regex);
28        Regex::new(&rust_pattern)
29            .with_context(|| format!("Failed to compile regex: {}", self.regex))
30    }
31
32    /// Get a display string showing the pattern and match count
33    pub fn display(&self) -> String {
34        format!("[{} matches] {}", self.match_count, self.regex)
35    }
36
37    /// Get the regex in modern format (for display purposes)
38    pub fn modern_format(&self) -> String {
39        regex_conversion::posix_to_modern(&self.regex)
40    }
41}
42
43/// Group similar log messages and generate regex patterns
44///
45/// # Arguments
46/// * `entries` - Log messages to analyze
47/// * `min_group_size` - Minimum number of messages to form a group
48///
49/// # Returns
50/// Vector of pattern groups, sorted by match count (descending)
51pub fn group_and_generate(entries: &[String], min_group_size: usize) -> Result<Vec<PatternGroup>> {
52    if entries.is_empty() {
53        return Ok(Vec::new());
54    }
55
56    // First pass: Group by exact prefix (first 50 chars or first word)
57    let mut prefix_groups: HashMap<String, Vec<usize>> = HashMap::new();
58
59    for (idx, entry) in entries.iter().enumerate() {
60        let prefix = extract_prefix(entry);
61        prefix_groups.entry(prefix).or_default().push(idx);
62    }
63
64    // Second pass: Generate regex for each group
65    let mut pattern_groups = Vec::new();
66
67    for (_prefix, indices) in prefix_groups {
68        // Skip small groups
69        if indices.len() < min_group_size {
70            continue;
71        }
72
73        // Get the actual log messages for this group
74        let group_messages: Vec<&str> = indices.iter().map(|&idx| entries[idx].as_str()).collect();
75
76        // Generate regex using grex (produces modern format with \d, \w, etc.)
77        let modern_regex = RegExpBuilder::from(&group_messages)
78            .with_conversion_of_digits()
79            .with_conversion_of_words()
80            .with_conversion_of_repetitions()
81            .build();
82
83        // Convert to POSIX format for logcheck compatibility
84        let posix_regex = regex_conversion::modern_to_posix(&modern_regex);
85
86        // Verify the regex matches all group members
87        // We need to convert back to Rust format for validation
88        let rust_regex = regex_conversion::posix_to_rust(&posix_regex);
89        if let Ok(compiled) = Regex::new(&rust_regex) {
90            let valid_indices: Vec<usize> = indices
91                .into_iter()
92                .filter(|&idx| compiled.is_match(&entries[idx]))
93                .collect();
94
95            if valid_indices.len() >= min_group_size {
96                pattern_groups.push(PatternGroup {
97                    regex: posix_regex,
98                    matching_indices: valid_indices.clone(),
99                    match_count: valid_indices.len(),
100                });
101            }
102        }
103    }
104
105    // Sort by match count (descending)
106    pattern_groups.sort_by(|a, b| b.match_count.cmp(&a.match_count));
107
108    Ok(pattern_groups)
109}

111/// Extract a representative prefix from a log message for grouping
112fn extract_prefix(entry: &str) -> String {
113    // Try to extract the first meaningful part
114    // This helps group similar messages together
115
116    // Strategy 1: Find the last colon in the first 80 chars (likely the service/program separator)
117    let search_limit = entry.len().min(80);
118    if let Some(colon_pos) = entry[..search_limit].rfind(':') {
119        return entry[..=colon_pos].to_string();
120    }
121
122    // Strategy 2: Take first N characters (up to 50)
123    let prefix_len = entry.len().min(50);
124    entry[..prefix_len].to_string()
125}

144    #[test]
145    fn test_group_and_generate() -> Result<()> {
146        let entries = vec![
147            "pam_unix(sudo:session): session opened for user root".to_string(),
148            "pam_unix(sudo:session): session closed for user root".to_string(),
149            "pam_unix(sudo:session): session opened for user alice".to_string(),
150            "systemd[1]: Started service foo".to_string(),
151            "systemd[1]: Started service bar".to_string(),
152        ];
153
154        let groups = group_and_generate(&entries, 2)?;
155
156        // Should have 2 groups: pam_unix (3 matches) and systemd (2 matches)
157        assert_eq!(groups.len(), 2, "Expected 2 groups, got {}", groups.len());
158
159        // First group should have most matches
160        assert_eq!(groups[0].match_count, 3);
161        // The regex is in POSIX format with word character classes
162        assert!(
163            groups[0].regex.contains("[[:alnum:]_]"),
164            "Expected POSIX word character class in: {}",
165            groups[0].regex
166        );
167
168        // Second group
169        assert_eq!(groups[1].match_count, 2);
170        assert!(
171            groups[1].regex.contains("[[:alnum:]_]") || groups[1].regex.contains("[[:digit:]]"),
172            "Expected character classes in: {}",
173            groups[1].regex
174        );
175
176        // Test that patterns can be compiled and match the original entries
177        let compiled1 = groups[0].compile()?;
178        assert!(compiled1.is_match(&entries[0]));
179        assert!(compiled1.is_match(&entries[1]));
180        assert!(compiled1.is_match(&entries[2]));
181
182        let compiled2 = groups[1].compile()?;
183        assert!(compiled2.is_match(&entries[3]));
184        assert!(compiled2.is_match(&entries[4]));
185
186        Ok(())
187    }
